// ───────────────────────────────────────────── // STATS / TRACKER MODULE — СТРОИТЕЛЬ.ПРО // Standalone analytics functions. Pure, no DOM. // Pass in the app state S (see DEF_STATE structure). // ───────────────────────────────────────────── function statEntEarn(S, e){ const b=(S.buildings||[]).find(x=>x.id===e.buildingId); const wt=(b?.workTypes||[]).find(w=>w.id===e.workTypeId); return Number(e.volume||0)*(wt?.price||0); } // Daily series for last N days: [{date, vol, earn, cum}] function statDailySeries(S, days=14){ const out=[];const now=new Date(); let cum=0; for(let i=days-1;i>=0;i--){ const d=new Date(now);d.setDate(d.getDate()-i); const dt=d.toISOString().slice(0,10); const ents=(S.entries||[]).filter(e=>e.dateFrom===dt); const vol=ents.reduce((s,e)=>s+Number(e.volume||0),0); const earn=ents.reduce((s,e)=>s+statEntEarn(S,e),0); cum+=earn; out.push({date:dt, vol:+vol.toFixed(2), earn:Math.round(earn), cum:Math.round(cum)}); } return out; } // Totals + averages (skips Sundays for day-average) function statSummary(S){ const ents=S.entries||[]; const totalVol=ents.reduce((s,e)=>s+Number(e.volume||0),0); const totalEarn=ents.reduce((s,e)=>s+statEntEarn(S,e),0); const workDays={}; ents.forEach(e=>{const d=new Date(e.dateFrom+'T12:00');if(d.getDay()!==0)workDays[e.dateFrom]=1;}); const dCnt=Object.keys(workDays).length||1; return { totalVol:+totalVol.toFixed(1), totalEarn:Math.round(totalEarn), netEarn:Math.round(totalEarn*0.87), // minus 13% tax workDays:dCnt, avgVolPerDay:+(totalVol/dCnt).toFixed(1), avgEarnPerDay:Math.round(totalEarn/dCnt), entryCount:ents.length, }; } // Per-worker productivity (requires e.worker + S.users) function statByWorker(S){ return (S.users||[]).map(u=>{ const ents=(S.entries||[]).filter(e=>e.worker===u.id); const days={};ents.forEach(e=>{const d=new Date(e.dateFrom+'T12:00');if(d.getDay()!==0)days[e.dateFrom]=1;}); const dCnt=Object.keys(days).length||1; const vol=ents.reduce((s,e)=>s+Number(e.volume||0),0); const earn=ents.reduce((s,e)=>s+statEntEarn(S,e),0); return {id:u.id, name:u.name, vol:+vol.toFixed(1), earn:Math.round(earn), dayVol:+(vol/dCnt).toFixed(1), entries:ents.length}; }).filter(w=>w.entries>0).sort((a,b)=>b.dayVol-a.dayVol); } // Per-workType progress for a building function statWtProgress(S, buildingId){ const b=(S.buildings||[]).find(x=>x.id===buildingId);if(!b)return []; return (b.workTypes||[]).map(wt=>{ const done=(S.entries||[]).filter(e=>e.buildingId===buildingId&&e.workTypeId===wt.id) .reduce((s,e)=>s+Number(e.volume||0),0); const target=Number(wt.targetArea)||0; return {name:wt.name, done:+done.toFixed(1), target, pct:target?Math.min(100,Math.round(done/target*100)):null, left:Math.max(0,+(target-done).toFixed(1)), earnLeft:Math.round(Math.max(0,target-done)*wt.price)}; }); } // Period stats (day/week/month) — REAL data, calendar-based function statPeriod(S, period='month'){ const now=new Date();const td=now.toISOString().slice(0,10); const inP=(dt)=>{ if(period==='day')return dt===td; const d=new Date(dt+'T12:00'); if(period==='week'){const diff=(now-d)/864e5;return diff>=0&&diff<7;} if(period==='month')return d.getMonth()===now.getMonth()&&d.getFullYear()===now.getFullYear(); return true; }; const ents=(S.entries||[]).filter(e=>inP(e.dateFrom)); const earn=ents.reduce((s,e)=>s+statEntEarn(S,e),0); return {period, vol:+ents.reduce((s,e)=>s+Number(e.volume||0),0).toFixed(1), earn:Math.round(earn), net:Math.round(earn*0.87), entries:ents.length}; } // Export for module usage if(typeof module!=='undefined'&&module.exports){ module.exports={statEntEarn,statDailySeries,statSummary,statByWorker,statWtProgress,statPeriod}; }
